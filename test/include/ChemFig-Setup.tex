\usepackage{chemfig}

\usetikzlibrary{decorations}
\usetikzlibrary{decorations.pathmorphing}

\pgfdeclaredecoration{hdashed}{initial} {
  \state{initial}[width=4pt] {
      \pgfpathmoveto{\pgfpoint{2pt}{0pt}}
      \pgfpathlineto{\pgfpoint{3pt}{0pt}}
      \pgfpathmoveto{\pgfpoint{4pt}{0pt}}}
  \state{final} {
     \pgfpathmoveto{\pgfpointdecoratedpathlast}} }

\tikzset{hbond/.style={decorate,decoration=hdashed}}

\pgfdeclaredecoration{empty}{initial} {
  \state{initial}[width=4pt] {
      % \pgfpathmoveto{\pgfpoint{4pt}{0pt}}
  }
  \state{final} {
     \pgfpathmoveto{\pgfpointdecoratedpathlast}} }

\tikzset{emptybond/.style={decorate,decoration=empty}}

\pgfdeclaredecoration{ddbond}{initial} {
   \state{initial}[width=4pt] {
      \pgfpathlineto{\pgfpoint{4pt}{0pt}}
      \pgfpathmoveto{\pgfpoint{2pt}{3pt}}
      \pgfpathlineto{\pgfpoint{4pt}{3pt}}
      \pgfpathmoveto{\pgfpoint{4pt}{0pt}} }
   \state{final} {
      \pgfpathlineto{\pgfpointdecoratedpathlast} } }

\tikzset{lddbond/.style={decorate,decoration=ddbond}}
\tikzset{rddbond/.style={decorate,decoration={ddbond,mirror}}}
